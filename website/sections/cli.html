Python orchestration

Build hybrid automations that shell out to the CLI, verify telemetry, and publish runbooks from a single script.

Test & attest

Configure CLI sandboxes to emit signed attestations so your compliance bot can ingest them instantly. 

from sapl.cli import command
from sapl.telemetry import capture

@command.pipeline()
def release(fleet: str):
    command.run(["sapl", "lint", f"manifests/{fleet}.yaml"], capture="json")
    capture.metric("lint.ok", value=True)

    with command.section("Render docs"):
        command.run(["sapl", "docs", "render", "docs.yaml", "--format", "portal"])
        command.run(["sapl", "publish", "docs", "--channel", "hologram"])

    command.run(["sapl", "timeline", "capture", fleet, "--label", "release"], capture="md")

if __name__ == "__main__":
    command.launch(release)
